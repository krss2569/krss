การตั้งค่า Google Apps Script

เพื่อบันทึกข้อมูลและรูปภาพลง Google Sheet และ Google Drive ให้ทำตามขั้นตอนดังนี้:

  1. สร้าง Google Sheet ใหม่ สร้างคอลัมน์ A-J: Timestamp, คำนำหน้า, ชื่อ, นามสกุล, ตำแหน่ง, โรงเรียน, สหวิทยาเขต, อีเมล, หมายเหตุ, สลิปโอนเงิน
  2. ไปที่เมนู ส่วนขยาย (Extensions) -> Apps Script
  3. ลบโค้ดเดิมออก และวางโค้ดด้านล่างนี้ลงไป:

const FOLDER_ID = 'ใส่_ID_โฟลเดอร์_Google_Drive_ที่นี่'; // สร้างโฟลเดอร์ใน Drive แล้วเอา ID มาใส่
const SHEET_NAME = 'Sheet1'; // ชื่อ Sheet ที่ต้องการบันทึก

function doPost(e) {
  try {
    let data = JSON.parse(e.postData.contents);
    let sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
    
    // จัดการไฟล์รูปภาพ Base64
    let fileUrl = "";
    if (data.fileBase64 && data.fileName) {
      let decodedData = Utilities.base64Decode(data.fileBase64.split(",")[1]);
      let blob = Utilities.newBlob(decodedData, data.fileMimeType, data.fileName);
      let folder = DriveApp.getFolderById(FOLDER_ID);
      let file = folder.createFile(blob);
      fileUrl = file.getUrl();
    }

    // บันทึกข้อมูลลง Sheet
    sheet.appendRow([
      new Date(),
      data.prefix,
      data.firstName,
      data.lastName,
      data.position,
      data.school,
      data.network,
      data.email,
      data.remark,
      fileUrl
    ]);

    return ContentService.createTextOutput(JSON.stringify({ status: 'success' })).setMimeType(ContentService.MimeType.JSON);
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ status: 'error', message: error.toString() })).setMimeType(ContentService.MimeType.JSON);
  }
}
